منصة عقارك - Mansat Aqaark

منصة عقارية شاملة للتحكم في العقارات والوسطاء مع لوحة تحكم متكاملة للمسؤول.

المكونات الرئيسية:

  • واجهة المستخدم المبنية بلغة React
  • الخادم والواجهة الخلفية المبنية بلغة Node.js (Express)
  • قاعدة بيانات تخزين مؤقت للتطوير
  • نظام إدارة متكامل متعدد المستويات

الأدوار في النظام

الدور الصلاحيات
المسؤول (Admin)
  • إدارة كاملة للنظام
  • الموافقة على حسابات الوسطاء
  • تمييز العقارات وإضافة العروض
  • إدارة المستخدمين وتغيير كلمات المرور
  • مراقبة النظام من خلال لوحة المعلومات
الوسيط (Broker)
  • إدارة العقارات الخاصة به
  • إضافة وتعديل وحذف العقارات
  • عرض المبيعات والأداء
  • تحديث الملف الشخصي
الزائر
  • استعراض العقارات
  • البحث والتصفية
  • التواصل مع الوسطاء
  • مشاهدة تفاصيل العقارات

export const AuthProvider: React.FC<{ children: React.ReactNode }> = ({ children }) => {
  const [authData, setAuthData] = useLocalStorage<{
    user: User | null;
    token: string | null;
  }>('auth', { user: null, token: null });
  
  const [isLoading, setIsLoading] = useState(false);
  const [error, setError] = useState(null);
  const { toast } = useToast();

  // Login function
  const login = async (credentials: LoginCredentials) => {
    setIsLoading(true);
    setError(null);
    
    try {
      const data = await loginUser(credentials);
      setAuthData({ user: data.user, token: data.token });
      toast({
        title: "تسجيل الدخول بنجاح",
        description: `مرحباً، ${data.user.name}!`,
      });
    } catch (err: any) {
      const errorMessage = err.message || 'فشل تسجيل الدخول';
      setError(errorMessage);
      toast({
        variant: "destructive",
        title: "خطأ في تسجيل الدخول",
        description: errorMessage,
      });
      throw err;
    } finally {
      setIsLoading(false);
    }
  };

  // Logout function
  const logout = () => {
    setAuthData({ user: null, token: null });
    toast({
      title: "تم تسجيل الخروج",
      description: "نأمل بزيارتك مرة أخرى قريباً!",
    });
  };

  // Add token to all requests
  useEffect(() => {
    if (authData.token) {
      localStorage.setItem('token', authData.token);
    } else {
      localStorage.removeItem('token');
    }
  }, [authData.token]);

  return (
    
      {children}
    
  );
};

API Routes (routes.ts)

export async function registerRoutes(app: Express): Promise {
  // Authentication routes
  app.post("/api/auth/login", async (req, res) => {
    try {
      const { username, password } = req.body;
      
      if (!username || !password) {
        return res.status(400).json({ message: "اسم المستخدم وكلمة المرور مطلوبة" });
      }
      
      const user = await storage.getUserByUsername(username);
      
      if (!user || user.password !== password) {
        return res.status(401).json({ message: "معلومات الدخول غير صحيحة" });
      }
      
      if (user.role === UserRole.BROKER && !user.approved) {
        return res.status(403).json({ message: "الحساب في انتظار الموافقة" });
      }
      
      // Generate JWT token
      const token = jwt.sign(
        { id: user.id, role: user.role, name: user.name },
        JWT_SECRET,
        { expiresIn: "24h" }
      );
      
      res.json({
        token,
        user: {
          id: user.id,
          name: user.name,
          username: user.username,
          phone: user.phone,
          role: user.role,
          properties: user.properties,
          sales: user.sales,
          profileImage: user.profileImage,
          licenseNumber: user.licenseNumber
        }
      });
    } catch (error) {
      res.status(500).json({ message: "Server error" });
    }
  });

  // Property routes
  app.get("/api/properties", async (req, res) => {
    try {
      const { type, category, status, featured, search } = req.query;
      
      const properties = await storage.getProperties({
        type: type as string,
        category: category as string,
        status: status as string,
        featured: featured === "true",
        search: search as string
      });
      
      res.json(properties);
    } catch (error) {
      res.status(500).json({ message: "Server error" });
    }
  });

  // Admin routes
  app.get("/api/admin/users", authenticate, authorize([UserRole.ADMIN]), async (_req, res) => {
    try {
      const users = await storage.getUsers();
      res.json(users);
    } catch (error) {
      res.status(500).json({ message: "Server error" });
    }
  });

  // User management routes
  app.put("/api/admin/users/:id/password", authenticate, authorize([UserRole.ADMIN]), async (req, res) => {
    try {
      const userId = parseInt(req.params.id);
      const { newPassword } = req.body;
      
      if (!newPassword || newPassword.length < 6) {
        return res.status(400).json({ message: "كلمة المرور يجب أن تكون 6 أحرف على الأقل" });
      }
      
      const user = await storage.getUser(userId);
      
      if (!user) {
        return res.status(404).json({ message: "المستخدم غير موجود" });
      }
      
      const updatedUser = await storage.updatePassword(userId, newPassword);
      
      // لا نرسل كلمة المرور في الرد
      if (updatedUser) {
        const { password, ...userWithoutPassword } = updatedUser;
        res.json({
          message: "تم تحديث كلمة المرور بنجاح",
          user: userWithoutPassword
        });
      } else {
        res.status(500).json({ message: "حدث خطأ أثناء تحديث كلمة المرور" });
      }
    } catch (error) {
      res.status(500).json({ message: "Server error" });
    }
  });

  return httpServer;
}
